def extract_features(destination_name):
   
    import re
    match = re.match(r'([^_]+)_([^_]+)_(.+?)\s\((\w+)\)', destination_name)

        place_code = match.group(2) + '_' + match.group(3)  
        state = match.group(4)  
        return city, place_code, state
    else:
        return None, None, None

def extract_features(source_name):
    import re
    match = re.match(r'([^_]+)_([^_]+)_(.+?)\s\((\w+)\)', source_name)
    if match:
        city = match.group(1) 
        place_code = match.group(2) + '_' + match.group(3)  
        state = match.group(4)  # Enclosed in parentheses
        return city, place_code, state
    else:
        return None, None, None

# Function to detect outliers using IQR
def detect_outliers_iqr(df):
    Q1 = df.quantile(0.25)
    Q3 = df.quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    outliers = (df < lower_bound) | (df > upper_bound)
    return outliers
